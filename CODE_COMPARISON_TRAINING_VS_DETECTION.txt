@app.route('/api/train', methods=['POST'])
def train():
    # 1ï¸âƒ£ Load data WITH labels
    train_loader, val_loader, test_loader, input_dim, num_classes = create_dataloaders(
        dataset_name=dataset_name,
        batch_size=batch_size,
        seq_len=seq_len
    )
    # â†‘ This expects data to have 'label' column!
    
    # 2ï¸âƒ£ Create model
    model = NextGenIDS(input_size=input_dim, ...).to(device)
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    criterion = torch.nn.CrossEntropyLoss()
    
    # 3ï¸âƒ£ TRAINING LOOP - Model LEARNS!
    for epoch in range(epochs):
        
        model.train()  # â† LEARNING MODE! Enables dropout, batch norm updates
        
        for X, y in train_loader:  # â† y = labels (the answers!)
            X, y = X.to(device), y.to(device)
            
            optimizer.zero_grad()  # â† Clear gradients
            
            logits = model(X)  # â† Make prediction
            
            loss = criterion(logits, y)  # â† Compare with CORRECT answer!
                                          # "How wrong am I?"
            
            loss.backward()  # â† Calculate gradients
                            # "Which neurons need adjusting?"
            
            optimizer.step()  # â† UPDATE WEIGHTS!
                             # "Make neurons smarter!"
        
        # Validation (also uses labels)
        model.eval()
        with torch.no_grad():
            for X, y in val_loader:  # â† y = validation labels
                logits = model(X)
                # Check accuracy, but DON'T learn from validation
    
    # 4ï¸âƒ£ Save the LEARNED model
    torch.save({
        'state_dict': model.state_dict(),  # â† Save learned weights
        'f1': best_f1
    }, checkpoint_path)
    
    return "Model trained successfully!"

@app.route('/api/evaluate', methods=['POST'])
def evaluate():
    # 1ï¸âƒ£ Load data (may or may not have labels)
    _, _, test_loader, input_dim, num_classes = create_dataloaders(
        dataset_name=dataset_name,
        batch_size=batch_size,
        seq_len=seq_len
    )
    # â†‘ If no labels, that's OK! We'll predict them
    
    # 2ï¸âƒ£ Load TRAINED model from checkpoint
    checkpoint_path = os.path.join(CHECKPOINT_FOLDER, checkpoint_name)
    ckpt = torch.load(checkpoint_path, map_location=device)
    
    # Detect which architecture (NextGenIDS or IDSModel)
    uses_arnn = any('arnn' in key for key in ckpt['state_dict'].keys())
    if uses_arnn:
        model = NextGenIDS(input_size=input_dim, ...).to(device)
    else:
        model = IDSModel(input_size=input_dim, ...).to(device)
    
    model.load_state_dict(ckpt['state_dict'])  # â† Load learned weights
    
    # 3ï¸âƒ£ PREDICTION LOOP - Model PREDICTS (NO LEARNING!)
    
    model.eval()  # â† PREDICTION MODE! Freeze dropout, batch norm
    
    test_preds, test_labels = [], []
    
    with torch.no_grad():  # â† CRITICAL! Disable gradient calculation
                          # "Don't learn, just predict!"
        
        for X, y in test_loader:  # â† y might not exist (or we ignore it)
            X, y = X.to(device), y.to(device)
            
            # NO optimizer.zero_grad()
            
            logits = model(X)  # â† Make prediction
            
            # NO loss calculation!
            # NO loss.backward()!
            # NO optimizer.step()!
            
            test_preds.extend(logits.argmax(1).cpu().numpy())  # â† Just save predictions
            test_labels.extend(y.cpu().numpy())  # â† Save labels (if exist, for comparison)
    
    # 4ï¸âƒ£ Calculate metrics (if labels available)
    if test_labels:
        metrics = compute_metrics(test_labels, test_preds)
    
    # 5ï¸âƒ£ Return predictions + metrics
    return jsonify({
        "predictions": test_preds,
        "metrics": metrics
    })
